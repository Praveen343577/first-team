    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
    <script>
        // Scene setup
        const scene = new THREE.Scene();
        
        const camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 1000);
        camera.position.set(0, 0, 8);
        
        const renderer = new THREE.WebGLRenderer({ 
            antialias: true, 
            alpha: true,
            powerPreference: "high-performance"
        });
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
        renderer.toneMapping = THREE.ACESFilmicToneMapping;
        renderer.toneMappingExposure = 1.5;
        document.getElementById('canvas-container').appendChild(renderer.domElement);
        
        // Holographic flowing gradient shader
        const holographicShader = {
            vertexShader: `
                varying vec2 vUv;
                varying vec3 vNormal;
                varying vec3 vPosition;
                varying float vFlow;
                
                uniform float uTime;
                uniform float uSpeed;
                uniform float uFlowIntensity;
                
                void main() {
                    vUv = uv;
                    vNormal = normalize(normalMatrix * normal);
                    vPosition = position;
                    
                    // Create flowing effect along the torus tube
                    float flow = sin(uv.x * 6.28318 * 3.0 + uTime * uSpeed) * 
                                 cos(uv.y * 6.28318 * 2.0 + uTime * uSpeed * 0.5);
                    vFlow = flow;
                    
                    // Slight vertex displacement for liquid feel
                    vec3 newPosition = position + normal * flow * 0.02 * uFlowIntensity;
                    
                    gl_Position = projectionMatrix * modelViewMatrix * vec4(newPosition, 1.0);
                }
            `,
            fragmentShader: `
                varying vec2 vUv;
                varying vec3 vNormal;
                varying vec3 vPosition;
                varying float vFlow;
                
                uniform float uTime;
                uniform float uSpeed;
                uniform vec3 uColorCyan;
                uniform vec3 uColorMagenta;
                uniform vec3 uColorOrange;
                uniform vec3 uColorBlue;
                uniform float uOpacity;
                uniform float uGlowIntensity;
                
                void main() {
                    // Fresnel for edge glow
                    vec3 viewDir = normalize(cameraPosition - vPosition);
                    float fresnel = pow(1.0 - abs(dot(viewDir, vNormal)), 2.0);
                    
                    // Create swirling holographic gradients
                    float angle = atan(vNormal.y, vNormal.x);
                    float swirl = sin(angle * 3.0 + uTime * uSpeed + vFlow * 2.0) * 0.5 + 0.5;
                    
                    // Mix the Chroma colors based on position and flow
                    vec3 color;
                    float t = vUv.x * 4.0 + uTime * uSpeed * 0.3;
                    
                    // Cyan to Magenta transition
                    vec3 colorCM = mix(uColorCyan, uColorMagenta, sin(t) * 0.5 + 0.5);
                    // Add Orange highlights
                    color = mix(colorCM, uColorOrange, pow(swirl, 3.0) * 0.7);
                    // Deep blue shadows
                    color = mix(uColorBlue, color, 0.3 + 0.7 * swirl);
                    
                    // Speed streaks - horizontal motion blur effect
                    float streaks = sin(vUv.x * 100.0 + uTime * uSpeed * 10.0) * 0.5 + 0.5;
                    streaks = pow(streaks, 20.0) * uSpeed * 0.3;
                    color += vec3(streaks);
                    
                    // Fresnel glow boost
                    color *= (1.0 + fresnel * uGlowIntensity);
                    
                    // Alpha with fresnel falloff
                    float alpha = uOpacity * (0.4 + fresnel * 0.6);
                    
                    gl_FragColor = vec4(color, alpha);
                }
            `
        };
        
        // Create multiple concentric tori like the image
        const toriGroup = new THREE.Group();
        scene.add(toriGroup);
        
        const torusConfigs = [
            { radius: 2.0, tube: 0.6, colorShift: 0.0, opacity: 0.9, glow: 2.0 },
            { radius: 2.3, tube: 0.15, colorShift: 0.3, opacity: 0.7, glow: 3.0 },
            { radius: 1.7, tube: 0.12, colorShift: 0.6, opacity: 0.6, glow: 2.5 },
            { radius: 2.6, tube: 0.08, colorShift: 0.9, opacity: 0.5, glow: 4.0 },
            { radius: 1.4, tube: 0.06, colorShift: 1.2, opacity: 0.4, glow: 3.5 }
        ];
        
        const tori = [];
        
        torusConfigs.forEach((config, index) => {
            const geometry = new THREE.TorusGeometry(config.radius, config.tube, 64, 200);
            const material = new THREE.ShaderMaterial({
                vertexShader: holographicShader.vertexShader,
                fragmentShader: holographicShader.fragmentShader,
                uniforms: {
                    uTime: { value: 0 },
                    uSpeed: { value: 1.0 },
                    uFlowIntensity: { value: 1.0 },
                    uColorCyan: { value: new THREE.Color(0x00d4ff) },
                    uColorMagenta: { value: new THREE.Color(0xff00ff) },
                    uColorOrange: { value: new THREE.Color(0xff8800) },
                    uColorBlue: { value: new THREE.Color(0x1a0a3e) },
                    uOpacity: { value: config.opacity },
                    uGlowIntensity: { value: config.glow }
                },
                transparent: true,
                side: THREE.DoubleSide,
                blending: THREE.AdditiveBlending,
                depthWrite: false
            });
            
            const torus = new THREE.Mesh(geometry, material);
            
            // Tilt like in the image - facing user but angled
            torus.rotation.x = Math.PI / 6; // 30 degree tilt
            torus.rotation.y = Math.PI / 8; // Slight Y tilt
            
            toriGroup.add(torus);
            tori.push({ mesh: torus, config: config, baseRotation: { x: Math.PI / 6, y: Math.PI / 8 } });
        });
        
        // Speed settings
        const speedSettings = {
            cruise: { rpm: 2000, multiplier: 0.5, flow: 1.0 },
            sport: { rpm: 4500, multiplier: 1.5, flow: 1.5 },
            race: { rpm: 8000, multiplier: 3.0, flow: 2.0 },
            boost: { rpm: 12000, multiplier: 6.0, flow: 3.0 }
        };
        
        let currentSpeed = 'cruise';
        let targetSpeed = speedSettings.cruise;
        let currentMultiplier = 0.5;
        let currentFlow = 1.0;
        let currentRpm = 2000;
        
        function setSpeed(mode) {
            currentSpeed = mode;
            targetSpeed = speedSettings[mode];
            
            document.querySelectorAll('.control-btn').forEach(btn => btn.classList.remove('active'));
            document.getElementById(`btn-${mode}`).classList.add('active');
        }
        
        window.setSpeed = setSpeed;
        
        // Mouse interaction for subtle parallax
        let mouseX = 0;
        let mouseY = 0;
        
        document.addEventListener('mousemove', (e) => {
            mouseX = (e.clientX / window.innerWidth - 0.5) * 2;
            mouseY = (e.clientY / window.innerHeight - 0.5) * 2;
        });
        
        // Animation loop
        const clock = new THREE.Clock();
        
        function animate() {
            requestAnimationFrame(animate);
            
            const delta = clock.getDelta();
            const elapsedTime = clock.getElapsedTime();
            
            // Smooth speed transitions
            currentMultiplier += (targetSpeed.multiplier - currentMultiplier) * delta * 3;
            currentFlow += (targetSpeed.flow - currentFlow) * delta * 3;
            currentRpm += (targetSpeed.rpm - currentRpm) * delta * 4;
            
            // Update UI
            document.getElementById('rpm').textContent = Math.round(currentRpm).toLocaleString();
            document.getElementById('rpm-bar').style.width = `${(currentRpm / 12000) * 100}%`;
            
            // Racing tire rotation - spinning like a wheel facing the viewer
            tori.forEach((item, index) => {
                const torus = item.mesh;
                const config = item.config;
                
                // Update shader uniforms
                torus.material.uniforms.uTime.value = elapsedTime;
                torus.material.uniforms.uSpeed.value = currentMultiplier;
                torus.material.uniforms.uFlowIntensity.value = currentFlow;
                
                // Racing rotation - spinning around Z axis (like a tire rolling toward you)
                // Each ring rotates at slightly different speed for parallax effect
                const direction = index % 2 === 0 ? 1 : -1;
                const speedVariation = 1 + index * 0.1;
                torus.rotation.z -= currentMultiplier * delta * speedVariation * direction;
                
                // Subtle wobble at high speeds (tire vibration)
                if (currentMultiplier > 2) {
                    const vibration = (currentMultiplier - 2) * 0.005;
                    torus.rotation.x = item.baseRotation.x + (Math.random() - 0.5) * vibration;
                    torus.rotation.y = item.baseRotation.y + (Math.random() - 0.5) * vibration;
                } else {
                    // Smooth return to base tilt
                    torus.rotation.x += (item.baseRotation.x - torus.rotation.x) * delta * 5;
                    torus.rotation.y += (item.baseRotation.y - torus.rotation.y) * delta * 5;
                }
            });
            
            // Group parallax based on mouse
            toriGroup.rotation.x = mouseY * 0.1;
            toriGroup.rotation.y = mouseX * 0.1;
            
            // Camera shake at extreme speeds
            if (currentMultiplier > 4) {
                camera.position.x = (Math.random() - 0.5) * 0.03 * (currentMultiplier - 4);
                camera.position.y = (Math.random() - 0.5) * 0.03 * (currentMultiplier - 4);
            } else {
                camera.position.x += (0 - camera.position.x) * delta * 10;
                camera.position.y += (0 - camera.position.y) * delta * 10;
            }
            
            renderer.render(scene, camera);
        }
        
        // Handle resize
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });
        
        animate();
    </script>
