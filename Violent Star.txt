    <script>
        // Scene setup
        const scene = new THREE.Scene();
        scene.fog = new THREE.FogExp2(0x000000, 0.02);
        
        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        camera.position.z = 5;
        
        const renderer = new THREE.WebGLRenderer({ 
            antialias: true, 
            alpha: true,
            powerPreference: "high-performance"
        });
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
        renderer.toneMapping = THREE.ACESFilmicToneMapping;
        renderer.toneMappingExposure = 1.2;
        document.getElementById('canvas-container').appendChild(renderer.domElement);
        
        // Holographic shader material
        const holographicVertexShader = `
            varying vec2 vUv;
            varying vec3 vNormal;
            varying vec3 vPosition;
            varying float vDistortion;
            
            uniform float uTime;
            uniform float uSpeed;
            uniform float uDistortion;
            
            // Simplex noise function
            vec3 mod289(vec3 x) { return x - floor(x * (1.0 / 289.0)) * 289.0; }
            vec4 mod289(vec4 x) { return x - floor(x * (1.0 / 289.0)) * 289.0; }
            vec4 permute(vec4 x) { return mod289(((x*34.0)+1.0)*x); }
            vec4 taylorInvSqrt(vec4 r) { return 1.79284291400159 - 0.85373472095314 * r; }
            
            float snoise(vec3 v) {
                const vec2 C = vec2(1.0/6.0, 1.0/3.0);
                const vec4 D = vec4(0.0, 0.5, 1.0, 2.0);
                
                vec3 i  = floor(v + dot(v, C.yyy));
                vec3 x0 = v - i + dot(i, C.xxx);
                
                vec3 g = step(x0.yzx, x0.xyz);
                vec3 l = 1.0 - g;
                vec3 i1 = min(g.xyz, l.zxy);
                vec3 i2 = max(g.xyz, l.zxy);
                
                vec3 x1 = x0 - i1 + C.xxx;
                vec3 x2 = x0 - i2 + C.yyy;
                vec3 x3 = x0 - D.yyy;
                
                i = mod289(i);
                vec4 p = permute(permute(permute(
                    i.z + vec4(0.0, i1.z, i2.z, 1.0))
                    + i.y + vec4(0.0, i1.y, i2.y, 1.0))
                    + i.x + vec4(0.0, i1.x, i2.x, 1.0));
                    
                float n_ = 0.142857142857;
                vec3 ns = n_ * D.wyz - D.xzx;
                
                vec4 j = p - 49.0 * floor(p * ns.z * ns.z);
                
                vec4 x_ = floor(j * ns.z);
                vec4 y_ = floor(j - 7.0 * x_);
                
                vec4 x = x_ *ns.x + ns.yyyy;
                vec4 y = y_ *ns.x + ns.yyyy;
                vec4 h = 1.0 - abs(x) - abs(y);
                
                vec4 b0 = vec4(x.xy, y.xy);
                vec4 b1 = vec4(x.zw, y.zw);
                
                vec4 s0 = floor(b0)*2.0 + 1.0;
                vec4 s1 = floor(b1)*2.0 + 1.0;
                vec4 sh = -step(h, vec4(0.0));
                
                vec4 a0 = b0.xzyw + s0.xzyw*sh.xxyy;
                vec4 a1 = b1.xzyw + s1.xzyw*sh.zzww;
                
                vec3 p0 = vec3(a0.xy, h.x);
                vec3 p1 = vec3(a0.zw, h.y);
                vec3 p2 = vec3(a1.xy, h.z);
                vec3 p3 = vec3(a1.zw, h.w);
                
                vec4 norm = taylorInvSqrt(vec4(dot(p0,p0), dot(p1,p1), dot(p2,p2), dot(p3,p3)));
                p0 *= norm.x;
                p1 *= norm.y;
                p2 *= norm.z;
                p3 *= norm.w;
                
                vec4 m = max(0.6 - vec4(dot(x0,x0), dot(x1,x1), dot(x2,x2), dot(x3,x3)), 0.0);
                m = m * m;
                return 42.0 * dot(m*m, vec4(dot(p0,x0), dot(p1,x1), dot(p2,x2), dot(p3,x3)));
            }
            
            void main() {
                vUv = uv;
                vNormal = normalize(normalMatrix * normal);
                
                // Create flowing noise based on rotation speed
                float noise = snoise(position * 2.0 + uTime * uSpeed * 0.5);
                float noise2 = snoise(position * 4.0 - uTime * uSpeed * 0.3);
                
                // Distortion increases with speed (tire deformation effect)
                float distortionStrength = uDistortion * (1.0 + uSpeed * 0.5);
                vec3 distorted = position + normal * (noise * 0.1 + noise2 * 0.05) * distortionStrength;
                
                vDistortion = noise;
                vPosition = distorted;
                
                gl_Position = projectionMatrix * modelViewMatrix * vec4(distorted, 1.0);
            }
        `;
        
        const holographicFragmentShader = `
            varying vec2 vUv;
            varying vec3 vNormal;
            varying vec3 vPosition;
            varying float vDistortion;
            
            uniform float uTime;
            uniform float uSpeed;
            uniform vec3 uColor1;
            uniform vec3 uColor2;
            uniform vec3 uColor3;
            uniform float uOpacity;
            
            void main() {
                // Fresnel effect for holographic look
                vec3 viewDirection = normalize(cameraPosition - vPosition);
                float fresnel = pow(1.0 - dot(viewDirection, vNormal), 3.0);
                
                // Create chromatic aberration based on speed
                float aberration = uSpeed * 0.02;
                float r = fresnel * (1.0 + aberration);
                float b = fresnel * (1.0 - aberration);
                
                // Holographic color mixing
                float t = uTime * 0.5 + vDistortion;
                vec3 color1 = uColor1; // Cyan
                vec3 color2 = uColor2; // Magenta
                vec3 color3 = uColor3; // Orange
                
                // Speed-based color intensity
                float speedBoost = 1.0 + uSpeed * 0.3;
                
                // Mix colors based on position and time
                vec3 finalColor = mix(
                    mix(color1, color2, sin(t) * 0.5 + 0.5),
                    color3,
                    cos(t * 1.3) * 0.5 + 0.5
                );
                
                // Add speed streaks
                float streaks = sin(vUv.y * 50.0 + uTime * uSpeed * 10.0) * 0.5 + 0.5;
                streaks = pow(streaks, 20.0) * uSpeed * 0.5;
                finalColor += vec3(streaks);
                
                // Fresnel glow
                finalColor *= (1.0 + fresnel * 2.0 * speedBoost);
                
                // Alpha based on fresnel for edge glow
                float alpha = fresnel * uOpacity * (0.8 + uSpeed * 0.2);
                
                gl_FragColor = vec4(finalColor * speedBoost, alpha);
            }
        `;
        
        // Create main orb
        const orbGeometry = new THREE.SphereGeometry(2, 128, 128);
        const orbMaterial = new THREE.ShaderMaterial({
            vertexShader: holographicVertexShader,
            fragmentShader: holographicFragmentShader,
            uniforms: {
                uTime: { value: 0 },
                uSpeed: { value: 1.0 },
                uDistortion: { value: 0.3 },
                uColor1: { value: new THREE.Color(0x00d4ff) }, // Cyan
                uColor2: { value: new THREE.Color(0xff00ff) }, // Magenta
                uColor3: { value: new THREE.Color(0xff8800) }, // Orange
                uOpacity: { value: 0.9 }
            },
            transparent: true,
            side: THREE.DoubleSide,
            blending: THREE.AdditiveBlending,
            depthWrite: false
        });
        
        const orb = new THREE.Mesh(orbGeometry, orbMaterial);
        scene.add(orb);
        
        // Create inner glow core
        const coreGeometry = new THREE.SphereGeometry(1.8, 64, 64);
        const coreMaterial = new THREE.ShaderMaterial({
            vertexShader: holographicVertexShader,
            fragmentShader: holographicFragmentShader,
            uniforms: {
                uTime: { value: 0 },
                uSpeed: { value: 1.0 },
                uDistortion: { value: 0.1 },
                uColor1: { value: new THREE.Color(0x0088ff) },
                uColor2: { value: new THREE.Color(0xff0088) },
                uColor3: { value: new THREE.Color(0xff4400) },
                uOpacity: { value: 0.4 }
            },
            transparent: true,
            side: THREE.BackSide,
            blending: THREE.AdditiveBlending,
            depthWrite: false
        });
        
        const core = new THREE.Mesh(coreGeometry, coreMaterial);
        orb.add(core);
        
        // Create outer energy ring (tire-like structure)
        const ringGeometry = new THREE.TorusGeometry(2.5, 0.05, 16, 100);
        const ringMaterial = new THREE.ShaderMaterial({
            vertexShader: `
                varying vec2 vUv;
                varying float vSpeed;
                uniform float uSpeed;
                uniform float uTime;
                
                void main() {
                    vUv = uv;
                    vSpeed = uSpeed;
                    
                    // Stretch the ring based on speed (motion blur effect)
                    vec3 pos = position;
                    float stretch = 1.0 + uSpeed * 0.1;
                    pos.x *= stretch;
                    
                    gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
                }
            `,
            fragmentShader: `
                varying vec2 vUv;
                varying float vSpeed;
                uniform float uTime;
                uniform vec3 uColor;
                
                void main() {
                    float streak = sin(vUv.x * 100.0 - uTime * vSpeed * 20.0) * 0.5 + 0.5;
                    streak = pow(streak, 10.0) * vSpeed;
                    
                    vec3 color = uColor * (1.0 + streak * 2.0);
                    float alpha = 0.6 + streak * 0.4;
                    
                    gl_FragColor = vec4(color, alpha);
                }
            `,
            uniforms: {
                uTime: { value: 0 },
                uSpeed: { value: 1.0 },
                uColor: { value: new THREE.Color(0x00d4ff) }
            },
            transparent: true,
            blending: THREE.AdditiveBlending,
            depthWrite: false
        });
        
        const ring = new THREE.Mesh(ringGeometry, ringMaterial);
        ring.rotation.x = Math.PI / 2;
        scene.add(ring);
        
        // Particle system for speed trails
        const particleCount = 2000;
        const particleGeometry = new THREE.BufferGeometry();
        const positions = new Float32Array(particleCount * 3);
        const velocities = new Float32Array(particleCount * 3);
        const colors = new Float32Array(particleCount * 3);
        
        for (let i = 0; i < particleCount; i++) {
            const i3 = i * 3;
            const angle = Math.random() * Math.PI * 2;
            const radius = 2.5 + Math.random() * 2;
            
            positions[i3] = Math.cos(angle) * radius;
            positions[i3 + 1] = (Math.random() - 0.5) * 4;
            positions[i3 + 2] = Math.sin(angle) * radius;
            
            velocities[i3] = (Math.random() - 0.5) * 0.1;
            velocities[i3 + 1] = (Math.random() - 0.5) * 0.1;
            velocities[i3 + 2] = (Math.random() - 0.5) * 0.1;
            
            // Holographic colors
            const colorChoice = Math.random();
            if (colorChoice < 0.33) {
                colors[i3] = 0.0; colors[i3 + 1] = 0.83; colors[i3 + 2] = 1.0; // Cyan
            } else if (colorChoice < 0.66) {
                colors[i3] = 1.0; colors[i3 + 1] = 0.0; colors[i3 + 2] = 1.0; // Magenta
            } else {
                colors[i3] = 1.0; colors[i3 + 1] = 0.53; colors[i3 + 2] = 0.0; // Orange
            }
        }
        
        particleGeometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
        particleGeometry.setAttribute('color', new THREE.BufferAttribute(colors, 3));
        
        const particleMaterial = new THREE.PointsMaterial({
            size: 0.03,
            vertexColors: true,
            transparent: true,
            opacity: 0.8,
            blending: THREE.AdditiveBlending,
            depthWrite: false
        });
        
        const particles = new THREE.Points(particleGeometry, particleMaterial);
        scene.add(particles);
        
        // Motion blur trail effect using multiple orb copies
        const trailCount = 5;
        const trails = [];
        
        for (let i = 0; i < trailCount; i++) {
            const trailOrb = new THREE.Mesh(
                new THREE.SphereGeometry(2 - i * 0.05, 64, 64),
                new THREE.ShaderMaterial({
                    vertexShader: holographicVertexShader,
                    fragmentShader: holographicFragmentShader,
                    uniforms: {
                        uTime: { value: 0 },
                        uSpeed: { value: 1.0 },
                        uDistortion: { value: 0.2 - i * 0.03 },
                        uColor1: { value: new THREE.Color(0x00d4ff) },
                        uColor2: { value: new THREE.Color(0xff00ff) },
                        uColor3: { value: new THREE.Color(0xff8800) },
                        uOpacity: { value: 0.3 - i * 0.05 }
                    },
                    transparent: true,
                    side: THREE.DoubleSide,
                    blending: THREE.AdditiveBlending,
                    depthWrite: false
                })
            );
            trailOrb.visible = false;
            scene.add(trailOrb);
            trails.push({ mesh: trailOrb, delay: (i + 1) * 0.02 });
        }
        
        // Speed settings
        const speedSettings = {
            cruise: { rpm: 2000, multiplier: 1, distortion: 0.2 },
            sport: { rpm: 4500, multiplier: 2.5, distortion: 0.4 },
            race: { rpm: 8000, multiplier: 5, distortion: 0.7 },
            boost: { rpm: 12000, multiplier: 10, distortion: 1.2 }
        };
        
        let currentSpeed = 'cruise';
        let targetSpeed = speedSettings.cruise;
        let currentMultiplier = 1;
        let currentDistortion = 0.2;
        let currentRpm = 2000;
        
        function setSpeed(mode) {
            currentSpeed = mode;
            targetSpeed = speedSettings[mode];
            
            // Update UI
            document.querySelectorAll('.control-btn').forEach(btn => btn.classList.remove('active'));
            document.getElementById(`btn-${mode}`).classList.add('active');
        }
        
        window.setSpeed = setSpeed;
        
        // Mouse interaction
        let mouseX = 0;
        let mouseY = 0;
        let targetRotationX = 0;
        let targetRotationY = 0;
        
        document.addEventListener('mousemove', (e) => {
            mouseX = (e.clientX / window.innerWidth) * 2 - 1;
            mouseY = -(e.clientY / window.innerHeight) * 2 + 1;
        });
        
        // Animation loop
        const clock = new THREE.Clock();
        const trailPositions = [];
        
        function animate() {
            requestAnimationFrame(animate);
            
            const delta = clock.getDelta();
            const elapsedTime = clock.getElapsedTime();
            
            // Smooth speed transitions
            currentMultiplier += (targetSpeed.multiplier - currentMultiplier) * delta * 2;
            currentDistortion += (targetSpeed.distortion - currentDistortion) * delta * 2;
            currentRpm += (targetSpeed.rpm - currentRpm) * delta * 3;
            
            // Update UI
            document.getElementById('rpm').textContent = Math.round(currentRpm).toLocaleString();
            document.getElementById('rpm-bar').style.width = `${(currentRpm / 12000) * 100}%`;
            
            // Racing tire rotation - fast spin on Z axis (like a rolling tire)
            const rotationSpeed = currentMultiplier * 2;
            orb.rotation.z -= rotationSpeed * delta;
            orb.rotation.x = Math.sin(elapsedTime * 0.5) * 0.1;
            
            // Update uniforms
            orbMaterial.uniforms.uTime.value = elapsedTime;
            orbMaterial.uniforms.uSpeed.value = currentMultiplier;
            orbMaterial.uniforms.uDistortion.value = currentDistortion;
            
            coreMaterial.uniforms.uTime.value = elapsedTime;
            coreMaterial.uniforms.uSpeed.value = currentMultiplier;
            
            ringMaterial.uniforms.uTime.value = elapsedTime;
            ringMaterial.uniforms.uSpeed.value = currentMultiplier;
            ring.rotation.z -= rotationSpeed * delta * 1.2;
            
            // Camera shake at high speeds
            if (currentMultiplier > 5) {
                camera.position.x = (Math.random() - 0.5) * 0.02 * (currentMultiplier - 5);
                camera.position.y = (Math.random() - 0.5) * 0.02 * (currentMultiplier - 5);
            } else {
                camera.position.x = 0;
                camera.position.y = 0;
            }
            
            // Particle animation - speed trails
            const positions = particles.geometry.attributes.position.array;
            for (let i = 0; i < particleCount; i++) {
                const i3 = i * 3;
                
                // Particles trail behind based on speed
                positions[i3] -= velocities[i3] * currentMultiplier;
                positions[i3 + 1] -= velocities[i3 + 1] * currentMultiplier * 0.5;
                positions[i3 + 2] -= velocities[i3 + 2] * currentMultiplier;
                
                // Reset particles that go too far
                const dist = Math.sqrt(
                    positions[i3] ** 2 + 
                    positions[i3 + 1] ** 2 + 
                    positions[i3 + 2] ** 2
                );
                
                if (dist > 6 || Math.random() < 0.01) {
                    const angle = Math.random() * Math.PI * 2;
                    const radius = 2.5;
                    positions[i3] = Math.cos(angle) * radius;
                    positions[i3 + 1] = (Math.random() - 0.5) * 2;
                    positions[i3 + 2] = Math.sin(angle) * radius;
                }
            }
            particles.geometry.attributes.position.needsUpdate = true;
            particles.rotation.y -= rotationSpeed * delta * 0.5;
            
            // Motion blur trails
            if (currentMultiplier > 3) {
                trailPositions.push(orb.rotation.z);
                if (trailPositions.length > trailCount) trailPositions.shift();
                
                trails.forEach((trail, i) => {
                    if (trailPositions.length > i) {
                        trail.mesh.visible = true;
                        trail.mesh.rotation.z = trailPositions[Math.max(0, trailPositions.length - 1 - i)] || orb.rotation.z;
                        trail.mesh.material.uniforms.uTime.value = elapsedTime - trail.delay;
                        trail.mesh.material.uniforms.uSpeed.value = currentMultiplier;
                        trail.mesh.scale.setScalar(1 - i * 0.05);
                    }
                });
            } else {
                trails.forEach(trail => trail.mesh.visible = false);
            }
            
            // Mouse tilt
            targetRotationX = mouseY * 0.2;
            targetRotationY = mouseX * 0.2;
            orb.rotation.x += (targetRotationX - orb.rotation.x) * delta * 2;
            orb.rotation.y += (targetRotationY - orb.rotation.y) * delta * 2;
            
            renderer.render(scene, camera);
        }
        
        // Handle resize
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });
        
        animate();
    </script>
